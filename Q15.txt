#include<stdio.h>
#include<math.h>
//Ler três valores e colocá-los em ordem crescente.
int main(void)
{
  float a,b,c,delta,x1,x2;
  printf(" digite o 'a' da funcao:");
  scanf("%f", &a);
  printf(" digite o 'b' da funcao:");
  scanf("%f", & b);
  printf(" digite o c da funcao:");
  scanf("%f", &c);
  delta= (b*b)-(4*a*c);
  printf("\n o delta é: %4.2f\n", delta);
  if(delta>0){
    x1= (-b+sqrt(delta))/(2*a);
    printf("\n x' equivale a:%4.2f\n", x1);
    x2= ((-b-sqrt(delta))/(2*a));
    printf("\n x'' equivale a:%4.2f\n", x2);
  }
  else{
    printf("\n n possui raiz real");
  }
  return 0;
}
